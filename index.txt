    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const menuItems = document.querySelectorAll('.menu-item');
            const summaryItemsDiv = document.getElementById('summary-items');
            const totalAmountSpan = document.getElementById('total-amount');
            const menuForm = document.getElementById('menuForm');

            function updateOrderSummary() {
                let total = 0;
                let selectedItems = []; // เก็บรายการที่เลือกสำหรับ localStorage
                summaryItemsDiv.innerHTML = ''; // Clear current summary

                menuItems.forEach(item => {
                    const itemName = item.dataset.name;
                    const itemPrice = parseFloat(item.dataset.price);
                    const quantityInput = item.querySelector('.item-qty');
                    const quantity = parseInt(quantityInput.value);

                    if (quantity > 0) {
                        const itemTotal = itemPrice * quantity;
                        total += itemTotal;

                        selectedItems.push({
                            name: itemName,
                            price: itemPrice,
                            quantity: quantity
                        });

                        const summaryItemDiv = document.createElement('div');
                        summaryItemDiv.classList.add('summary-item');
                        summaryItemDiv.innerHTML = `
                            <span>${itemName} x ${quantity}</span>
                            <span>${itemTotal} บาท</span>
                        `;
                        summaryItemsDiv.appendChild(summaryItemDiv);
                    }
                });
                totalAmountSpan.textContent = total;
                // บันทึกรายการที่เลือกและยอดรวมลงใน localStorage ทันทีที่มีการเปลี่ยนแปลง
                localStorage.setItem('selectedMenuItems', JSON.stringify(selectedItems));
                localStorage.setItem('totalOrderAmount', total);
            }

            menuItems.forEach(item => {
                const decreaseBtn = item.querySelector('.decrease-qty');
                const increaseBtn = item.querySelector('.increase-qty');
                const quantityInput = item.querySelector('.item-qty');

                decreaseBtn.addEventListener('click', () => {
                    let currentQty = parseInt(quantityInput.value);
                    if (currentQty > 0) {
                        quantityInput.value = currentQty - 1;
                        updateOrderSummary();
                    }
                });

                increaseBtn.addEventListener('click', () => {
                    let currentQty = parseInt(quantityInput.value);
                    quantityInput.value = currentQty + 1;
                    updateOrderSummary();
                });

                quantityInput.addEventListener('change', () => {
                    // Ensure quantity is not negative
                    if (parseInt(quantityInput.value) < 0 || isNaN(parseInt(quantityInput.value))) {
                        quantityInput.value = 0;
                    }
                    updateOrderSummary();
                });
            });

            // โหลดข้อมูลที่เลือกไว้ก่อนหน้าจาก localStorage (ถ้ามี)
            const savedItems = JSON.parse(localStorage.getItem('selectedMenuItems'));
            if (savedItems) {
                menuItems.forEach(item => {
                    const itemName = item.dataset.name;
                    const quantityInput = item.querySelector('.item-qty');
                    const foundItem = savedItems.find(saved => saved.name === itemName);
                    if (foundItem) {
                        quantityInput.value = foundItem.quantity;
                    }
                });
            }

            // Initial summary update
            updateOrderSummary();

            // เมื่อกดปุ่ม "ถัดไป" ในหน้าเมนู
            menuForm.addEventListener('submit', (event) => {
                event.preventDefault(); // ป้องกันการ submit form ปกติ

                const total = parseFloat(totalAmountSpan.textContent);
                if (total === 0) {
                    alert('กรุณาเลือกรายการอาหารก่อน');
                    return;
                }
                
                // ไปยังหน้าข้อมูลลูกค้า
                window.location.href = 'address.html';
            });
        });
    </script>
